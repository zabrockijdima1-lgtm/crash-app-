let multiplier = 1.00 let crashPoint = Math.random()*4+1.5 let running = true let playing = false let bet = 0 let balance = 100 const multiplierElement = document.getElementById("multiplier") function startRound(){ multiplier = 1 crashPoint = Math.random()*4+1.5 running = true playing = false gameLoop() } function gameLoop(){ if(!running) return multiplier += 0.02 multiplierElement.textContent = multiplier.toFixed(2)+"x" if(multiplier >= crashPoint){ running = false multiplierElement.textContent = "💥 "+multiplier.toFixed(2)+"x" multiplierElement.style.color = "#ff4444" setTimeout(startRound,3000) return } requestAnimationFrame(gameLoop) } function placeBet(amount){ if(playing) return if(balance < amount){ alert("Недостатньо коштів") return } bet = amount balance -= amount playing = true alert("Ставка "+amount+" TON") } function cashout(){ if(!playing) return if(!running) return let win = bet * multiplier balance += win playing = false alert("Ви забрали "+multiplier.toFixed(2)+"x") } document.querySelectorAll(".additional-item").forEach(item=>{ item.addEventListener("click",()=>{ let value = parseFloat(item.textContent) placeBet(value) }) }) startRound()
